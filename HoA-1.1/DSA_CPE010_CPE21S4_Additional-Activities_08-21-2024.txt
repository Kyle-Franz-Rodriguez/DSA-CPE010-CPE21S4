Additional Activity 2:

#Write an algorithm to read 100 numbers and then display the sum.

start program
	add variable "sum" = 0
	get input
	place input to variable "n"
	count the numbers of "n" and place to variable "total"
	add "n" to "sum"
	if "total" reached 100, end program
	else go back to get input
end program

Additional Activity 3:

#Write an algorithm to read two numbers then display the largest.

start program
	get 2 inputs
	place each input to variables "n1" and "n2"
	compare "n1" and "n2"
	if n1 > n2, print n1
	if n2 > n1, print n2
	if n1 = n2, print "neither"
end program

Additional Activity 5:

#Write an algorithm to read three numbers then display the largest

start program
	get 2 inputs
	place each input to variables "n1", "n2", "n3"
	compare "n1", "n2", "n3"
	if n1 > n2 && n3, print n1
	if n2 > n1 && n3, print n2
	if n3 > n1 && n2, print n3
	if n1 && n2 > n3, print "n1 and n2"
	if n1 && n3 > n2, print "n1 and n3"
	if n2 && n3 > n1, print "n2 and n3"
	if n1 = n2 = n3, print "none"
end program
